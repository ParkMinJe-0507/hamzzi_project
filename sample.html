class HeaderComponent extends HTMLElement { connectedCallback() { this.render(); this.addEventListeners(); this.loadTheme(); } render() { this.innerHTML = `

  • 홈
  • 햄스터 도감
    • 햄스터 종류
    • 먹이 정보

    • 건강 관리
  • 커뮤니티
    • 자유게시판
    • 정보 공유
    • 햄스터 자랑
  • 부가기능
    • 종합 맞춤 추천
    • 햄스터 병원 찾기
  • FAQ
로그인
`; } addEventListeners() { const searchBtn = this.querySelector('#search-btn'); const closeSearchBtn = this.querySelector('#close-search'); const searchOverlay = this.querySelector('#search-overlay'); const toggleBtn = this.querySelector('#toggle-mode'); const searchInput = this.querySelector('#search-input'); searchBtn.addEventListener('click', () => { searchOverlay.classList.add('is-active'); searchInput.focus(); }); closeSearchBtn.addEventListener('click', () => { searchOverlay.classList.remove('is-active'); }); searchOverlay.addEventListener('click', (e) => { if(e.target === searchOverlay) { searchOverlay.classList.remove('is-active'); } }); toggleBtn.addEventListener('click', () => { document.body.classList.toggle('dark'); localStorage.setItem('theme', document.body.classList.contains('dark') ? 'dark' : 'light'); this.updateThemeIcon(); }); searchInput.addEventListener('keydown', (event) => { if (event.key === 'Enter') { const query = searchInput.value.trim(); if (query) { window.location.href = `/search.html?q=${encodeURIComponent(query)}`; } } }); } updateThemeIcon() { const icon = this.querySelector('.mode-icon'); if (document.body.classList.contains('dark')) { icon.textContent = '🌙'; } else { icon.textContent = '☀️'; } } loadTheme() { if (localStorage.getItem('theme') === 'dark') { document.body.classList.add('dark'); } this.updateThemeIcon(); } } customElements.define('hamzzi-header', HeaderComponent);
